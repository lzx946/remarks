CREATE TABLE `t96365_mall`.`data_collect_log` (
  `log_id` varchar(64) NOT NULL,
  `pk_value` varchar(20) NOT NULL COMMENT '主键值',
  `data_type` varchar(20) NOT NULL COMMENT '数据类型',
  `data` text COMMENT '数据，json格式',
  `status` varchar(2) NOT NULL DEFAULT '0' COMMENT '同步状态，0=未同步，1=已同步',
  `synchronize_time` timestamp NULL DEFAULT NULL COMMENT '同步时间',
  `del_flag` char(1) NOT NULL DEFAULT '1' COMMENT '逻辑删除用',
  `create_id` varchar(20) DEFAULT NULL COMMENT '创建人ID',
  `modify_id` varchar(20) DEFAULT NULL COMMENT '修改人ID',
  `create_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `modify_date` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  PRIMARY KEY (`log_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='数据收集日志表';

CREATE TABLE `t96365_mall`.`data_syn_cursor` (
  `syn_id` varchar(20) NOT NULL COMMENT '同步ID',
  `log_id` varchar(64) NOT NULL COMMENT '数据变更日志ID',
  `channel_id` varchar(20) NOT NULL COMMENT '渠道ID',
  `channel_name` varchar(50) NOT NULL COMMENT '渠道名称',
  `status` varchar(2) NOT NULL DEFAULT '0' COMMENT '同步状态，0=未同步，1=已同步',
  `synchronize_time` timestamp NULL DEFAULT NULL COMMENT '同步时间',
  `del_flag` char(1) NOT NULL DEFAULT '1' COMMENT '逻辑删除用',
  `create_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `modify_date` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  PRIMARY KEY (`syn_id`),
	KEY (`log_id`),
	KEY (`channel_id`),
	KEY (`status`),
	KEY (`del_flag`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='同步日志游标表';
