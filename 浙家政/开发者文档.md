CREATE TABLE `developer` (
  `app_id` varchar(20) NOT NULL,
  `app_secret` varchar(100) NOT NULL,
  `company_name` varchar(100) NOT NULL COMMENT '公司名称',
  `application_name` varchar(100) NOT NULL COMMENT '平台(网站)名称',
  `application_url` varchar(255) NOT NULL COMMENT '平台(网站)网址',
  `email` varchar(100) NOT NULL COMMENT '邮箱',
  `contact_name` varchar(20) NOT NULL COMMENT '联系人',
  `contact_phone` varchar(20) NOT NULL COMMENT '联系电话',
  `business_license` varchar(255) NOT NULL COMMENT '营业执照',
  `public_key` text NOT NULL COMMENT '公钥',
  `private_key` text NOT NULL COMMENT '私钥',
  `status` varchar(20) NOT NULL COMMENT '状态',
  `audit_time` TIMESTAMP NULL DEFAULT NULL COMMENT '审核时间',
  `access_token` varchar(500) DEFAULT NULL COMMENT '授权token',
  `token_create_time` TIMESTAMP NULL DEFAULT NULL COMMENT '授权token生成时间',
  `del_flag` varchar(1) NOT NULL DEFAULT '1' COMMENT '逻辑删除用',
  `create_id` varchar(20) DEFAULT NULL COMMENT '创建人',
  `modify_id` varchar(20) DEFAULT NULL COMMENT '修改人',
  `create_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `modify_date` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  PRIMARY KEY (`app_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='开发者信息表';

-- 暂时不需要这张表
CREATE TABLE `developer_data_association` (
  `association_id` varchar(20) NOT NULL COMMENT '关联IDid',
  `association_type` varchar(20) NOT NULL COMMENT '关联类型',
  `app_id` varchar(50) NOT NULL COMMENT '开发者ID',
  `object_id` varchar(60) NOT NULL COMMENT '被关联表的ID',
  `identity_id` varchar(30) NOT NULL COMMENT '身份ID（身份证号、社会统一信用代码）',
  `del_flag` varchar(1) NOT NULL DEFAULT '1' COMMENT '逻辑删除标志',
  `create_date` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `modify_date` TIMESTAMP NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  PRIMARY KEY (`association_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='开发者数据关联表';



-- 订单新增表
CREATE TABLE `order_add` (
  `add_id` varchar(20) NOT NULL COMMENT '订单新增ID',
  `order_id` varchar(64) NOT NULL COMMENT '订单ID',
  `app_id` varchar(64) NOT NULL COMMENT '开发者ID',
  `province` varchar(64) NOT NULL COMMENT '省',
  `city` varchar(64) NOT NULL COMMENT '市',
  `district` varchar(64) NOT NULL COMMENT '区县',
  `area_name` varchar(64) NOT NULL COMMENT '小区或街道',
  `detail_addr` varchar(64) DEFAULT NULL COMMENT '详细地址',
  `longitude` varchar(64) DEFAULT NULL COMMENT '经度',
  `latitude` varchar(64) DEFAULT NULL COMMENT '纬度',
  `del_flag` varchar(1) NOT NULL DEFAULT '1' COMMENT '逻辑删除用',
  `create_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间' ,
  `modify_date` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  PRIMARY KEY (`add_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='订单新增信息表';


-- 订单服务人员关联表
CREATE TABLE `order_employee` (
  `order_emp_id` varchar(20) NOT NULL COMMENT '订单服务人员ID',
  `order_id` varchar(64) NOT NULL COMMENT '订单ID',
  `app_id` varchar(64) NOT NULL COMMENT '开发者ID',
  `emp_id` varchar(20) NOT NULL COMMENT '服务人员ID',
  `del_flag` varchar(1) NOT NULL DEFAULT '1' COMMENT '逻辑删除用',
  `create_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间' ,
  `modify_date` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  PRIMARY KEY (`order_emp_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='订单服务人员关联信息表';

-- 订单取消表
CREATE TABLE `order_cancel` (
  `cancel_id` varchar(20) NOT NULL COMMENT '订单服务人员ID',
  `order_id` varchar(64) NOT NULL COMMENT '订单ID',
  `app_id` varchar(64) NOT NULL COMMENT '开发者ID',
  `remark` varchar(200) NOT NULL COMMENT '备注',
  `del_flag` varchar(1) NOT NULL DEFAULT '1' COMMENT '逻辑删除用',
  `create_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间' ,
  `modify_date` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  PRIMARY KEY (`cancel_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='订单取消信息表';


-- 订单支付表
CREATE TABLE `order_pay` (
  `pay_id` varchar(20) NOT NULL COMMENT '订单支付ID',
  `order_id` varchar(64) NOT NULL COMMENT '订单ID',
  `app_id` varchar(64) NOT NULL COMMENT '开发者ID',
  `amount` int(10) NOT NULL COMMENT '订单金额，单位：分',
  `del_flag` varchar(1) NOT NULL DEFAULT '1' COMMENT '逻辑删除用',
  `create_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间' ,
  `modify_date` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  PRIMARY KEY (`pay_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='订单支付信息表';
