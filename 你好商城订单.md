```sql
ALTER table `order_refund` modify column `refund_status` varchar(1) NOT NULL DEFAULT '0' COMMENT '退款单状态（0=待处理；1=退款成功；2=已拒绝；3=已撤销；4=待买家发货；5=待卖家收货）';
```

### 订单在各个状态下的可操作行为

- 待付款 

> * 取消订单(买家)
> * 付款(买家)

- 待确认

> * 确认(**卖家**)
> * 取消订单(买家)

- 待发货

> * 修改地址(买家)
> * 退款(买家)
> * 批量退款(买家)
> * 确认发货(**卖家**)

- 待收货

> * 确认收货(买家)
> * 退换(买家)
> * 批量退换(买家)
> * 查看物流(买家)

- 交易成功

> * 评价(买家)
> * 删除订单(买家)
> * 退款(买家)
> * 退货退款(买家)
> * 查看物流(买家)

- 已取消

> * 删除订单(买家)

### 退款单在各个状态下的可操作行为

- 待处理
> * 取消退款(买家)
> * 同意退款(**卖家**)
> * 拒绝退款(**卖家**)

- 退款成功

> * 删除订单(买家)
> * 评价(买家)

- 已拒绝

- 已撤销


### 退货单在各个状态下的可操作行为

- 待处理

> * 取消退货(买家)
> * 同意退货(**卖家**)
> * 拒绝退货(**卖家**)

- 待买家发货

> * 取消退货(买家)

- 待卖家收货

> * 确认收货(**卖家**)

- 退货成功

- 已撤销
