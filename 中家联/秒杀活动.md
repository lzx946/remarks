## 建表

```sql
CREATE TABLE `seckill_quota_log` (
  `log_id` varchar(20) NOT NULL,
  `order_id` varchar(20) NOT NULL COMMENT '关联order_main表中order_id',
  `sku_id` varchar(20) NOT NULL COMMENT '商品规格子商品Id',
  `member_id` varchar(20) NOT NULL COMMENT '会员ID',
  `status` varchar(10) NOT NULL COMMENT '状态',
  `create_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `modify_date` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`log_id`),
  UNIQUE KEY `order_id` (`order_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='秒杀活动名额占用记录';
```
